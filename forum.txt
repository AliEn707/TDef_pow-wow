как могут быть модели
Forum  - раздел
has_many :topics, as: :topicable, dependent: :destroy
has_many :threads, dependent: :destroy
belongs_to :user
[
	name string
	description text
	user_id integer
	closed boolean
]

Forum::Thread - подраздел
has_many :topics, as: :topicable, dependent: :destroy
belongs_to :forum, touch: true
belongs_to :user
[
	name string
	description text
	user_id integer
	forum_id integer
	closed boolean
]
	
Forum::Topic - тема
has_many :messages, :dependent => :destroy#, :as =>:msg_dest, :class_name => "Message" #если Message который уже есть
belongs_to :topicable, polymorphic: true, touch: true
belongs_to :user
[
	name string
	description text
	user_id integer
	topicable_id integer
	topicable_type string
	closed boolean
]

сообщения или Forum::Message или Message
если Forum::Message
belongs_to :topic, class_name: "Forum::Topic"
belongs_to :user
[
	data text
	user_id integer
	topic_id integer
]

или Message уже есть
belongs_to :user, touch: true
belongs_to :msg_dest, polymorphic: true, touch: true
[
	data text
	user_id integer
	msg_dest_id integer
	msg_dest_type string
]
